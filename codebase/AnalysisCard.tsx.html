import { TouchCard } from '@/components/TouchCard';

import { CardContent, CardHeader, CardTitle } from '@/components/ui/card';

import { Badge } from '@/components/ui/badge';

import { Progress } from '@/components/ui/progress';

import { Startup } from '@/lib/mockData';

import { Building2, Calendar, MapPin, Users, DollarSign } from 'lucide-react';


interface AnalysisCardProps {

  startup: Startup;

  onClick?: () => void;

}


export default function AnalysisCard({ startup, onClick }: AnalysisCardProps) {

  const getRecommendationColor = (recommendation: string) => {

    switch (recommendation) {

      case 'Strong Buy': return 'bg-green-500 hover:bg-green-600';

      case 'Buy': return 'bg-blue-500 hover:bg-blue-600';

      case 'Hold': return 'bg-yellow-500 hover:bg-yellow-600';

      case 'Pass': return 'bg-red-500 hover:bg-red-600';

      default: return 'bg-gray-500 hover:bg-gray-600';

    }

  };


  const getRiskColor = (risk: string) => {

    switch (risk) {

      case 'Low': return 'bg-green-100 text-green-800';

      case 'Medium': return 'bg-yellow-100 text-yellow-800';

      case 'High': return 'bg-red-100 text-red-800';

      default: return 'bg-gray-100 text-gray-800';

    }

  };


  return (

    <TouchCard 

      interactive={true}

      onClick={onClick}

      className="min-h-[280px] p-1"

    >

      <CardHeader className="pb-4">

        <div className="flex justify-between items-start gap-4">

          <div className="min-w-0 flex-1">

            <CardTitle className="text-lg md:text-xl truncate">{startup.name}</CardTitle>

            <p className="text-sm text-muted-foreground mt-1 line-clamp-2">{startup.description}</p>

          </div>

          <div className="flex flex-col items-end gap-2 flex-shrink-0">

            <Badge className={`${getRecommendationColor(startup.recommendation)} text-xs px-2 py-1`}>

              {startup.recommendation}

            </Badge>

            <Badge variant="outline" className={`${getRiskColor(startup.riskLevel)} text-xs px-2 py-1`}>

              {startup.riskLevel} Risk

            </Badge>

          </div>

        </div>

      </CardHeader>

      

      <CardContent className="space-y-4 pt-0">

        <div className="flex items-center justify-between">

          <span className="text-sm font-medium">Overall Score</span>

          <span className="text-xl md:text-2xl font-bold">{startup.overallScore}/10</span>

        </div>

        

        <Progress value={startup.overallScore * 10} className="h-3" />

        

        <div className="grid grid-cols-2 gap-3 text-xs md:text-sm">

          <div className="space-y-2">

            <div className="flex items-center gap-2">

              <Building2 className="h-3 w-3 md:h-4 md:w-4 text-muted-foreground flex-shrink-0" />

              <span className="truncate">{startup.industry}</span>

            </div>

            <div className="flex items-center gap-2">

              <Calendar className="h-3 w-3 md:h-4 md:w-4 text-muted-foreground flex-shrink-0" />

              <span className="truncate">{startup.stage}</span>

            </div>

            <div className="flex items-center gap-2">

              <MapPin className="h-3 w-3 md:h-4 md:w-4 text-muted-foreground flex-shrink-0" />

              <span className="truncate">{startup.location}</span>

            </div>

          </div>

          

          <div className="space-y-2">

            <div className="flex items-center gap-2">

              <Users className="h-3 w-3 md:h-4 md:w-4 text-muted-foreground flex-shrink-0" />

              <span className="truncate">{startup.employees} employees</span>

            </div>

            <div className="flex items-center gap-2">

              <DollarSign className="h-3 w-3 md:h-4 md:w-4 text-muted-foreground flex-shrink-0" />

              <span className="truncate">${(startup.fundingRaised / 1000000).toFixed(1)}M raised</span>

            </div>

            <div className="flex items-center gap-2">

              <Calendar className="h-3 w-3 md:h-4 md:w-4 text-muted-foreground flex-shrink-0" />

              <span className="truncate">Founded {startup.foundedYear}</span>

            </div>

          </div>

        </div>

        

        <div className="pt-2 border-t">

          <p className="text-xs text-muted-foreground">

            Analyzed on {new Date(startup.analysisDate).toLocaleDateString()}

          </p>

        </div>

      </CardContent>

    </TouchCard>

  );

}

import { Startup } from './mockData';


export interface AnalysisInput {

  companyName: string;

  industry: string;

  stage: string;

  foundedYear: number;

  founders: string;

  description: string;

  website?: string;

  fundingRaised: number;

  employees: number;

  location: string;

  pitchDeck?: File;

  financialData?: File;

  additionalInfo?: string;

}


export const analyzeStartup = (input: AnalysisInput): Startup => {

  // Mock AI analysis - in real implementation, this would call AI APIs

  const marketScore = Math.random() * 3 + 7; // 7-10 range

  const teamScore = Math.random() * 3 + 6.5; // 6.5-9.5 range

  const productScore = Math.random() * 3 + 6; // 6-9 range

  const financialScore = Math.random() * 4 + 5; // 5-9 range

  

  const overallScore = (marketScore + teamScore + productScore + financialScore) / 4;

  

  // Determine risk level based on stage and scores

  let riskLevel: 'Low' | 'Medium' | 'High';

  if (input.stage === 'Series B' || input.stage === 'Series C') {

    riskLevel = overallScore > 8 ? 'Low' : 'Medium';

  } else if (input.stage === 'Series A') {

    riskLevel = overallScore > 7.5 ? 'Medium' : 'High';

  } else {

    riskLevel = overallScore > 8 ? 'Medium' : 'High';

  }

  

  // Determine recommendation

  let recommendation: 'Strong Buy' | 'Buy' | 'Hold' | 'Pass';

  if (overallScore >= 8.5) recommendation = 'Strong Buy';

  else if (overallScore >= 7.5) recommendation = 'Buy';

  else if (overallScore >= 6.5) recommendation = 'Hold';

  else recommendation = 'Pass';

  

  // Generate insights based on industry and stage

  const keyInsights = generateInsights(input, overallScore);

  const concerns = generateConcerns(input, riskLevel);

  

  return {

    id: Date.now().toString(),

    name: input.companyName,

    industry: input.industry,

    stage: input.stage,

    foundedYear: input.foundedYear,

    founders: input.founders.split(',').map(f => f.trim()),

    description: input.description,

    website: input.website,

    fundingRaised: input.fundingRaised,

    employees: input.employees,

    location: input.location,

    analysisDate: new Date().toISOString().split('T')[0],

    overallScore: Math.round(overallScore * 10) / 10,

    marketScore: Math.round(marketScore * 10) / 10,

    teamScore: Math.round(teamScore * 10) / 10,

    productScore: Math.round(productScore * 10) / 10,

    financialScore: Math.round(financialScore * 10) / 10,

    riskLevel,

    recommendation,

    keyInsights,

    concerns

  };

};


const generateInsights = (input: AnalysisInput, score: number): string[] => {

  const insights = [];

  

  if (score > 8) {

    insights.push('Strong overall performance across all metrics');

  }

  

  if (input.stage === 'Series A' || input.stage === 'Series B') {

    insights.push('Mature funding stage indicates proven business model');

  }

  

  if (input.fundingRaised > 10000000) {

    insights.push('Significant funding raised demonstrates investor confidence');

  }

  

  if (input.employees > 50) {

    insights.push('Substantial team size indicates scaling operations');

  }

  

  // Industry-specific insights

  if (input.industry.toLowerCase().includes('ai') || input.industry.toLowerCase().includes('artificial')) {

    insights.push('Operating in high-growth AI market with significant potential');

  }

  

  if (input.industry.toLowerCase().includes('health')) {

    insights.push('Healthcare sector offers large addressable market opportunities');

  }

  

  return insights.slice(0, 4); // Limit to 4 insights

};


const generateConcerns = (input: AnalysisInput, riskLevel: string): string[] => {

  const concerns = [];

  

  if (riskLevel === 'High') {

    concerns.push('High risk profile requires careful monitoring');

  }

  

  if (input.stage === 'Seed' || input.stage === 'Pre-Seed') {

    concerns.push('Early stage with limited market validation');

  }

  

  if (input.employees < 10) {

    concerns.push('Small team size may limit execution capacity');

  }

  

  if (input.fundingRaised < 1000000) {

    concerns.push('Limited funding may constrain growth opportunities');

  }

  

  // Industry-specific concerns

  if (input.industry.toLowerCase().includes('health')) {

    concerns.push('Healthcare regulations may impact time to market');

  }

  

  return concerns.slice(0, 3); // Limit to 3 concerns

};

import { Dialog, DialogContent, DialogDescription, DialogHeader, DialogTitle } from '@/components/ui/dialog';

import { TouchButton } from '@/components/TouchButton';

import { Badge } from '@/components/ui/badge';

import { Progress } from '@/components/ui/progress';

import { CheckCircle, ArrowRight, Zap } from 'lucide-react';


interface FeatureDetail {

  title: string;

  description: string;

  benefits: string[];

  technicalSpecs: string[];

  useCases: string[];

  metrics: { label: string; value: number }[];

}


interface FeatureDetailModalProps {

  isOpen: boolean;

  onClose: () => void;

  feature: FeatureDetail | null;

  onTryFeature?: () => void;

}


export default function FeatureDetailModal({ 

  isOpen, 

  onClose, 

  feature, 

  onTryFeature 

}: FeatureDetailModalProps) {

  if (!feature) return null;


  return (

    <Dialog open={isOpen} onOpenChange={onClose}>

      <DialogContent className="max-w-3xl max-h-[80vh] overflow-y-auto">

        <DialogHeader>

          <DialogTitle className="text-xl md:text-2xl flex items-center gap-2">

            <Zap className="h-6 w-6 text-blue-600" />

            {feature.title}

          </DialogTitle>

          <DialogDescription>

            {feature.description}

          </DialogDescription>

        </DialogHeader>

        

        <div className="space-y-6">

          {/* Benefits */}

          <div>

            <h3 className="text-lg font-semibold mb-3">Key Benefits</h3>

            <div className="grid gap-2">

              {feature.benefits.map((benefit, index) => (

                <div key={index} className="flex items-start gap-2">

                  <CheckCircle className="h-4 w-4 text-green-600 mt-1 flex-shrink-0" />

                  <span className="text-sm">{benefit}</span>

                </div>

              ))}

            </div>

          </div>

          

          {/* Technical Specifications */}

          <div>

            <h3 className="text-lg font-semibold mb-3">Technical Specifications</h3>

            <div className="grid gap-2">

              {feature.technicalSpecs.map((spec, index) => (

                <Badge key={index} variant="outline" className="justify-start text-xs">

                  {spec}

                </Badge>

              ))}

            </div>

          </div>

          

          {/* Use Cases */}

          <div>

            <h3 className="text-lg font-semibold mb-3">Use Cases</h3>

            <div className="grid gap-3 md:grid-cols-2">

              {feature.useCases.map((useCase, index) => (

                <div key={index} className="p-3 bg-gray-50 rounded-lg">

                  <p className="text-sm font-medium">Scenario {index + 1}</p>

                  <p className="text-xs text-muted-foreground mt-1">{useCase}</p>

                </div>

              ))}

            </div>

          </div>

          

          {/* Performance Metrics */}

          <div>

            <h3 className="text-lg font-semibold mb-3">Performance Metrics</h3>

            <div className="space-y-3">

              {feature.metrics.map((metric, index) => (

                <div key={index}>

                  <div className="flex justify-between mb-1">

                    <span className="text-sm font-medium">{metric.label}</span>

                    <span className="text-sm">{metric.value}%</span>

                  </div>

                  <Progress value={metric.value} className="h-2" />

                </div>

              ))}

            </div>

          </div>

        </div>

        

        <div className="flex gap-3 pt-4">

          {onTryFeature && (

            <TouchButton 

              onClick={onTryFeature}

              className="flex-1 bg-gradient-to-r from-blue-600 to-indigo-600 hover:from-blue-700 hover:to-indigo-700"

            >

              Try This Feature

              <ArrowRight className="h-4 w-4 ml-2" />

            </TouchButton>

          )}

          <TouchButton onClick={onClose} variant="outline">

            Close

          </TouchButton>

        </div>

      </DialogContent>

    </Dialog>

  );

}

import { Card } from '@/components/ui/card';

import { cn } from '@/lib/utils';

import { forwardRef } from 'react';


interface TouchCardProps extends React.HTMLAttributes<HTMLDivElement> {

  children: React.ReactNode;

  interactive?: boolean;

}


const TouchCard = forwardRef<HTMLDivElement, TouchCardProps>(

  ({ className, children, interactive = false, ...props }, ref) => {

    return (

      <Card

        ref={ref}

        className={cn(

          // Base touch-friendly styling

          'touch-manipulation select-none',

          // Interactive cards get enhanced touch feedback

          interactive && [

            'cursor-pointer transition-all duration-200',

            'hover:shadow-lg hover:scale-[1.02]',

            'active:scale-[0.98] active:shadow-md',

            'focus-visible:ring-2 focus-visible:ring-blue-500 focus-visible:outline-none',

            // Better touch targets

            'min-h-[60px]'

          ],

          className

        )}

        {...props}

      >

        {children}

      </Card>

    );

  }

);


TouchCard.displayName = 'TouchCard';


export { TouchCard };

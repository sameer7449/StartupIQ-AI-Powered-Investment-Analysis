import { useParams, useNavigate } from 'react-router-dom';

import { TouchButton } from '@/components/TouchButton';

import { TouchCard } from '@/components/TouchCard';

import { CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';

import { Badge } from '@/components/ui/badge';

import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';

import { mockStartups, Startup } from '@/lib/mockData';

import { 

  ArrowLeft, 

  Building2, 

  Calendar, 

  MapPin, 

  Users, 

  DollarSign, 

  Globe, 

  TrendingUp,

  AlertTriangle,

  CheckCircle,

  BarChart3,

  Share2,

  Download

} from 'lucide-react';


export default function StartupProfile() {

  const { id } = useParams<{ id: string }>();

  const navigate = useNavigate();

  const [startup, setStartup] = useState<Startup | null>(null);


  useEffect(() => {

    // Load startup from mock data or localStorage

    const storedStartups = JSON.parse(localStorage.getItem('analyzedStartups') || '[]');

    const allStartups = [...mockStartups, ...storedStartups];

    const foundStartup = allStartups.find(s => s.id === id);

    setStartup(foundStartup || null);

  }, [id]);

            </TouchButton>

          </CardContent>

        </TouchCard>

      </div>

    );

  }

    switch (recommendation) {

      case 'Strong Buy': return 'bg-green-500 hover:bg-green-600';

      case 'Buy': return 'bg-blue-500 hover:bg-blue-600';

      case 'Hold': return 'bg-yellow-500 hover:bg-yellow-600';

      default: return 'bg-gray-500 hover:bg-gray-600';

    }

  };

      case 'Medium': return 'bg-yellow-100 text-yellow-800';

      case 'High': return 'bg-red-100 text-red-800';

      default: return 'bg-gray-100 text-gray-800';

    }

            <TouchButton

              onClick={() => navigate('/startups')}

              </TouchButton>

            </div>

          </div>

        </div>

      </header>

                  {startup.industry}

                  <Calendar className="h-4 w-4" />

                  {startup.stage}

                </div>

                <div className="flex items-center gap-1">

                  <MapPin className="h-4 w-4" />

                  {startup.location}

                </div>

              </div>

            </div>

                <div className="text-3xl md:text-4xl font-bold">{startup.overallScore}/10</div>

              </div>

            </div>

          </div>

        </div>

                </CardHeader>

                    <div>

                      <div className="font-medium text-muted-foreground">Founded</div>

                        >

                      </div>

                    </div>

                  )}

                </CardContent>

              </TouchCard>

              <TouchCard className="border-0 shadow-lg bg-white/80 backdrop-blur-sm">

                <CardHeader>

                  <CardTitle>Founding Team</CardTitle>

                </CardHeader>

                <CardContent>

                </CardContent>

              </TouchCard>

            </div>

          </TabsContent>

          

          <TabsContent value="metrics" className="space-y-6">

            <div className="grid gap-6 md:grid-cols-2">

              {/* Score Breakdown */}

              <TouchCard className="border-0 shadow-lg bg-white/80 backdrop-blur-sm">

                <CardHeader>

                  <CardTitle className="flex items-center gap-2">

                    <BarChart3 className="h-5 w-5" />

                    Score Breakdown

                  </CardTitle>

                </CardHeader>

                <CardContent className="space-y-4">

                  <div className="space-y-4">

                    <div>

                      <div className="flex justify-between mb-2">

                        <span className="text-sm md:text-base font-medium">Market Score</span>

                        <span className="text-sm md:text-base">{startup.marketScore}/10</span>

                    </div>

                    

                    <div>

                      <div className="flex justify-between mb-2">

                        <span className="text-sm md:text-base font-medium">Team Score</span>

                        <span className="text-sm md:text-base">{startup.teamScore}/10</span>

                      </div>

                      <Progress value={startup.teamScore * 10} className="h-3" />

                    </div>

                    

                    <div>

                      <div className="flex justify-between mb-2">

                        <span className="text-sm md:text-base font-medium">Product Score</span>

                        <span className="text-sm md:text-base">{startup.productScore}/10</span>

                      </div>

                      <Progress value={startup.productScore * 10} className="h-3" />

                    </div>

                    

                    <div>

                      <div className="flex justify-between mb-2">

                        <span className="text-sm md:text-base font-medium">Financial Score</span>

                        <span className="text-sm md:text-base">{startup.financialScore}/10</span>

                      </div>

                      <Progress value={startup.financialScore * 10} className="h-3" />

                    </div>

                  </div>

                </CardContent>

              </TouchCard>


              {/* Overall Assessment */}

              <TouchCard className="border-0 shadow-lg bg-white/80 backdrop-blur-sm">

                <CardHeader>

                  <CardTitle>Overall Assessment</CardTitle>

                </CardHeader>

                <CardContent className="space-y-4">

                  <div className="text-center">

                    <div className="text-4xl md:text-5xl font-bold mb-4">{startup.overallScore}/10</div>

                    <Progress value={startup.overallScore * 10} className="h-4 mb-6" />

                    <Badge className={`${getRecommendationColor(startup.recommendation)} text-base px-4 py-2`}>

                      {startup.recommendation}

                    </Badge>

                  </div>

                  

                  <div className="pt-4 text-center">

                    <div className="text-sm text-muted-foreground mb-2">Risk Level</div>

                    <Badge variant="outline" className={`${getRiskColor(startup.riskLevel)} text-base px-3 py-1`}>

                      {startup.riskLevel} Risk

                  </div>

                </CardContent>

              </TouchCard>

            </div>

          </TabsContent>

                </CardHeader>

                    {startup.keyInsights.map((insight, index) => (

                </CardHeader>

                    {startup.concerns.map((concern, index) => (

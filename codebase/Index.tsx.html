import { useState } from 'react';

import { TouchButton } from '@/components/TouchButton';

import { TouchCard } from '@/components/TouchCard';

import FeatureCard from '@/components/FeatureCard';

import MarketDataModal from '@/components/MarketDataModal';

import FeatureDetailModal from '@/components/FeatureDetailModal';

import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';

import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';

import MetricsOverview from '@/components/MetricsOverview';

import AnalysisCard from '@/components/AnalysisCard';

import { mockStartups } from '@/lib/mockData';

import { marketSuccessStories, featureDetails } from '@/lib/marketData';

import { useNavigate } from 'react-router-dom';

import { Plus, Brain, TrendingUp, Shield, Zap, BarChart3 } from 'lucide-react';


interface MarketData {

  title: string;

  companies: Array<{

    name: string;

    industry: string;

    founded: number;

    valuation: string;

    employees: string;

    keyMetric: string;

    description: string;

    website?: string;

  }>;

}


interface FeatureDetail {

  title: string;

  description: string;

  benefits: string[];

  technicalSpecs: string[];

  useCases: string[];

  metrics: { label: string; value: number }[];

}


export default function Index() {

  const navigate = useNavigate();

  const [recentAnalyses] = useState(mockStartups.slice(0, 3));

  const [marketModalOpen, setMarketModalOpen] = useState(false);

  const [featureModalOpen, setFeatureModalOpen] = useState(false);

  const [selectedMarketData, setSelectedMarketData] = useState<MarketData>({title: '', companies: []});

  const [selectedFeature, setSelectedFeature] = useState<FeatureDetail | null>(null);


  const handleMarketData = (type: keyof typeof marketSuccessStories, title: string) => {

    setSelectedMarketData({ title, companies: marketSuccessStories[type] });

    setMarketModalOpen(true);

  };


  const handleFeatureDetail = (type: keyof typeof featureDetails) => {

    setSelectedFeature(featureDetails[type]);

    setFeatureModalOpen(true);

  };


  const handleTryFeature = (path: string) => {

    setFeatureModalOpen(false);

    navigate(path);

  };


  return (

    <div className="min-h-screen bg-gradient-to-br from-slate-50 via-blue-50 to-indigo-50">

      {/* Header */}

      <header className="border-b bg-white/80 backdrop-blur-sm sticky top-0 z-50">

        <div className="container mx-auto px-4 py-4">

          <div className="flex items-center justify-between">

            <div className="flex items-center space-x-3">

              <div className="h-10 w-10 bg-gradient-to-r from-blue-600 to-indigo-600 rounded-lg flex items-center justify-center">

                <Brain className="h-6 w-6 text-white" />

              </div>

              <div>

                <h1 className="text-2xl font-bold bg-gradient-to-r from-blue-600 to-indigo-600 bg-clip-text text-transparent">

                  StartupIQ

                </h1>

                <p className="text-sm text-muted-foreground">AI-Powered Investment Analysis</p>

              </div>

            </div>

            

            <div className="flex items-center space-x-2 md:space-x-4">

              <TouchButton 

                onClick={() => navigate('/startups')}

                variant="outline"

                className="hidden sm:flex"

              >

                View All Startups

              </TouchButton>

              <TouchButton 

                onClick={() => navigate('/upload')}

                className="bg-gradient-to-r from-blue-600 to-indigo-600 hover:from-blue-700 hover:to-indigo-700"

              >

                <Plus className="h-4 w-4 mr-2" />

                <span className="hidden sm:inline">Analyze Startup</span>

                <span className="sm:hidden">Analyze</span>

              </TouchButton>

            </div>

          </div>

        </div>

      </header>


      <main className="container mx-auto px-4 py-8">

        {/* Hero Section */}

        <div className="text-center mb-12">

          <h2 className="text-3xl md:text-4xl font-bold mb-4 bg-gradient-to-r from-gray-900 to-gray-600 bg-clip-text text-transparent">

            Transform Startup Analysis with AI

          </h2>

          <p className="text-lg md:text-xl text-muted-foreground max-w-3xl mx-auto mb-8">

            Synthesize founder materials and public data to generate actionable investment insights in minutes, not weeks.

          </p>

          

          <div className="grid md:grid-cols-3 gap-6 max-w-4xl mx-auto mb-12">

            <FeatureCard

              icon={<Zap className="h-12 w-12 mx-auto text-blue-600" />}

              title="Lightning Fast"

              description="Get comprehensive startup analysis in under 5 minutes"

              examples={["Stripe", "Vercel", "Algolia"]}

              onLearnMore={() => handleFeatureDetail('lightningFast')}

              onTryNow={() => navigate('/upload')}

            />

            

            <FeatureCard

              icon={<Brain className="h-12 w-12 mx-auto text-indigo-600" />}

              title="AI-Powered"

              description="Advanced algorithms analyze market, team, product, and financials"

              examples={["OpenAI", "Anthropic", "Scale AI"]}

              onLearnMore={() => handleFeatureDetail('aiPowered')}

              onTryNow={() => navigate('/upload')}

            />

            

            <FeatureCard

              icon={<Shield className="h-12 w-12 mx-auto text-green-600" />}

              title="Data-Driven"

              description="Objective insights based on comprehensive data synthesis"

              examples={["Palantir", "Snowflake", "Databricks"]}

              onLearnMore={() => handleFeatureDetail('dataDriven')}

              onTryNow={() => navigate('/startups')}

            />

          </div>


          {/* Quick Action Buttons */}

          <div className="flex flex-col sm:flex-row gap-4 justify-center items-center mb-8">

            <TouchButton 

              onClick={() => navigate('/upload')}

              size="lg"

              className="bg-gradient-to-r from-blue-600 to-indigo-600 hover:from-blue-700 hover:to-indigo-700 w-full sm:w-auto"

            >

              <Plus className="h-5 w-5 mr-2" />

              Start Analysis

            </TouchButton>

            <TouchButton 

              onClick={() => navigate('/startups')}

              variant="outline"

              size="lg"

              className="w-full sm:w-auto"

            >

              <TrendingUp className="h-5 w-5 mr-2" />

              View Portfolio

            </TouchButton>

          </div>

        </div>


        {/* Dashboard Content */}

        <Tabs defaultValue="overview" className="space-y-6">

          <TabsList className="grid w-full grid-cols-2 max-w-md mx-auto h-12">

            <TabsTrigger value="overview" className="text-sm md:text-base">Overview</TabsTrigger>

            <TabsTrigger value="recent" className="text-sm md:text-base">Recent Analyses</TabsTrigger>

          </TabsList>

          

          <TabsContent value="overview" className="space-y-6">

            <TouchCard 

              className="border-0 shadow-lg bg-white/80 backdrop-blur-sm"

              interactive={true}

              onClick={() => handleMarketData('platformMetrics', 'Platform Metrics')}

            >

              <CardHeader>

                <CardTitle className="flex items-center gap-2">

                  <TrendingUp className="h-5 w-5" />

                  Platform Metrics

                  <TouchButton 

                    variant="ghost" 

                    size="sm"

                    onClick={(e) => {

                      e.stopPropagation();

                      handleFeatureDetail('platformMetrics');

                    }}

                  >

                    View Details

                  </TouchButton>

                </CardTitle>

                <CardDescription>

                  Key performance indicators for your investment analysis platform

                </CardDescription>

              </CardHeader>

              <CardContent>

                <MetricsOverview />

                <div className="mt-4 flex gap-2">

                  <TouchButton 

                    variant="outline" 

                    size="sm"

                    onClick={(e) => {

                      e.stopPropagation();

                      handleMarketData('platformMetrics', 'Platform Metrics');

                    }}

                  >

                    <BarChart3 className="h-4 w-4 mr-1" />

                    Market Data

                  </TouchButton>

                </div>

              </CardContent>

            </TouchCard>

          </TabsContent>

          

          <TabsContent value="recent" className="space-y-6">

            <TouchCard className="border-0 shadow-lg bg-white/80 backdrop-blur-sm">

              <CardHeader>

                <CardTitle>Recent Startup Analyses</CardTitle>

                <CardDescription>

                  Latest startups analyzed by the AI engine

                </CardDescription>

              </CardHeader>

              <CardContent>

                <div className="grid gap-6 md:grid-cols-2 lg:grid-cols-3">

                  {recentAnalyses.map((startup) => (

                    <AnalysisCard 

                      key={startup.id} 

                      startup={startup}

                      onClick={() => navigate(`/startup/${startup.id}`)}

                    />

                  ))}

                </div>

                

                <div className="mt-6 text-center">

                  <TouchButton 

                    variant="outline" 

                    onClick={() => navigate('/startups')}

                  >

                    View All Analyses

                  </TouchButton>

                </div>

              </CardContent>

            </TouchCard>

          </TabsContent>

        </Tabs>

      </main>


      {/* Modals */}

      <MarketDataModal

        isOpen={marketModalOpen}

        onClose={() => setMarketModalOpen(false)}

        title={selectedMarketData.title}

        companies={selectedMarketData.companies}

      />


      <FeatureDetailModal

        isOpen={featureModalOpen}

        onClose={() => setFeatureModalOpen(false)}

        feature={selectedFeature}

        onTryFeature={() => {

          if (selectedFeature?.title.includes('Lightning Fast') || selectedFeature?.title.includes('AI-Powered')) {

            handleTryFeature('/upload');

          } else {

            handleTryFeature('/startups');

          }

        }}

      />

    </div>

  );

}
